# 2) Migrations SQL

> Fichier : `migrations.sql`

```sql
-- Database: message_anonyme
CREATE DATABASE IF NOT EXISTS message_anonyme CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci;
USE message_anonyme;

-- users (pour future intégration)
CREATE TABLE IF NOT EXISTS users (
  id INT AUTO_INCREMENT PRIMARY KEY,
  username VARCHAR(50) UNIQUE,
  email VARCHAR(255) UNIQUE,
  password_hash VARCHAR(255),
  display_name VARCHAR(100),
  accept_anonymous TINYINT(1) DEFAULT 1,
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  is_active TINYINT(1) DEFAULT 1
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- cards (cartes publiques qui reçoivent des messages)
CREATE TABLE IF NOT EXISTS cards (
  id INT AUTO_INCREMENT PRIMARY KEY,
  user_id INT,
  slug VARCHAR(120) NOT NULL UNIQUE,
  title VARCHAR(255),
  description TEXT,
  palette JSON,
  font VARCHAR(64),
  links JSON,
  allow_anonymous TINYINT(1) DEFAULT 1,
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE SET NULL
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- anonymous_messages (messages envoyés via une card)
CREATE TABLE IF NOT EXISTS anonymous_messages (
  id INT AUTO_INCREMENT PRIMARY KEY,
  card_id INT NOT NULL,
  pseudo VARCHAR(120) DEFAULT NULL,
  message_text TEXT DEFAULT NULL,
  image_path VARCHAR(500) NOT NULL,
  ip_hash VARCHAR(128) DEFAULT NULL,
  status ENUM('unread','read','replied','reported','deleted') DEFAULT 'unread',
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  FOREIGN KEY (card_id) REFERENCES cards(id) ON DELETE CASCADE,
  INDEX (card_id),
  INDEX (created_at)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- conversations (pour créer conversations anonymes)
CREATE TABLE IF NOT EXISTS conversations (
  id INT AUTO_INCREMENT PRIMARY KEY,
  card_id INT NOT NULL,
  participant_token VARCHAR(255) NOT NULL, -- token representing the sender (anon)
  is_open TINYINT(1) DEFAULT 1,
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  FOREIGN KEY (card_id) REFERENCES cards(id) ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE IF NOT EXISTS conversation_messages (
  id INT AUTO_INCREMENT PRIMARY KEY,
  conversation_id INT NOT NULL,
  from_role ENUM('user','anon') NOT NULL,
  body TEXT NOT NULL,
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  FOREIGN KEY (conversation_id) REFERENCES conversations(id) ON DELETE CASCADE,
  INDEX (conversation_id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- rate_limits (simple pour limiter les envois par IP)
CREATE TABLE IF NOT EXISTS rate_limits (
  id INT AUTO_INCREMENT PRIMARY KEY,
  ip_hash VARCHAR(128) NOT NULL,
  endpoint VARCHAR(255) NOT NULL,
  count INT DEFAULT 1,
  window_start TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  UNIQUE KEY unique_ip_endpoint (ip_hash, endpoint, window_start)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

Tu peux ajouter des index supplémentaires si tu prévois beaucoup de lecture sur anonymous_messages.
